pipeline {
    agent any

    environment {
        DOCKERHUB_CREDENTIALS = credentials('Docker')
    }


    stages {
        stage('Commit') {
            steps {
                sh 'sudo docker-compose down'
            }
        }
        
         stage('Deploy') {
            steps {
                sh 'sudo docker-compose up -d --force-recreate --no-deps --build test'
            }
        }

        
    }
}
