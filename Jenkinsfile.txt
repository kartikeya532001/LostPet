pipeline {
    agent any

    environment {
        DOCKERHUB_CREDENTIALS = credentials('kartikeya532001')
    }

    stages {
        stage('Commit') {
            steps {
                sh 'sudo docker-compose down'
            }
        }

        stage('Build') {
            steps {
                sh 'sudo docker build -t kartikeya532001/Frontend:1.0 .'
            }
        }

        stage('Login') {
            steps {
                sh 'echo $DOCKERHUB_CREDENTIALS_PSW | docker login -u $DOCKERHUB_CREDENTIALS_USR --password-stdin'
            }
        }

        stage('Push') {
            steps {
                sh 'sudo docker push kartikeya532001/Frontend:1.0'
            }
        }

        stage('deployment') {
            steps {
               sh 'sudo docker-compose up -d --force-recreate --no-deps --build test'
            }
        }
    }
}
