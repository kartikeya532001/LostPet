pipeline {
    agent any

    tools {
        nodejs 'Node'
    }


    stages {
        stage('clean') {
            steps {
                sh 'sudo docker-compose down'
            }
        }
        stage('Build') {
            steps {
               sh 'npm build'
            }
        }
         stage('Test') {
            steps {
               sh 'npm test'
            }
        }
        
        
         stage('Deploy') {
            steps {
               sh 'sudo docker-compose up -d --force-recreate --no-deps --build test'
            }
        }

      
    }
}
